\chapter{Poisson Infrastructure}

This chapter develops the measure-theoretic foundations for symmetric $\gamma$-stable
L\'evy processes. We define the stable L\'evy density $\nu_\gamma$, prove its symmetry
and moment properties, and establish the Poisson identities (mean and variance) that
underpin the obstruction argument. The chapter culminates in the canonical Poisson
construction: an explicit probability space $(\mathbb{N}, \mathrm{poissonMeasure}(\lambda T))$
on which the large-jump count $G_0$ is realized as a concrete random variable with
a proved (not axiomatized) Poisson distribution.

\section{Stable Lévy Measure}

The stable density $\nu_\gamma$ and its basic properties (symmetry, nonnegativity) are established here.

\begin{defn}[Stable Lévy density]\label{def:stable_density}
\lean{stableLevyDensity}
\leanok
The symmetric $\gamma$-stable Lévy measure density is
$\nu_\gamma(dz) = c_\gamma |z|^{-1-\gamma}\,dz$ on $\mathbb{R} \setminus \{0\}$,
where $c_\gamma > 0$ and $\gamma \in (1,2)$.
The singularity at the origin is strong enough that
$\int |z|^2\,\nu_\gamma(dz) = \infty$, forcing the energy space to be Banach rather than Hilbert.
This density is the starting point for the entire Lévy--Itô construction and all moment estimates that follow; see [4, 5] for background on stable processes.
\end{defn}

\begin{theorem}[Density symmetry]\label{thm:density_sym}
\lean{stableLevyDensity_symmetric}
\leanok
\uses{def:stable_density}
$\nu_\gamma(-z) = \nu_\gamma(z)$ for all $z \neq 0$, from the absolute value in $c_\gamma |z|^{-1-\gamma}$.
Symmetry of $\nu_\gamma$ implies $L \stackrel{d}{=} -L$, the key ingredient in the even-annihilation argument.
Without this property, the obstruction witness could not be shown to lie outside $\mathrm{im}(\delta_L)$.
\end{theorem}
\begin{proof}\leanok Unfold the definition and use $|-z| = |z|$. Closed by \texttt{simp [abs\_neg]}.\end{proof}

\begin{theorem}[Density nonnegativity]\label{thm:density_nonneg}
\lean{stableLevyDensity_nonneg}
\leanok
\uses{def:stable_density}
$c_\gamma |z|^{-1-\gamma} > 0$ for $z \neq 0$ when $c_\gamma > 0$, ensuring $\nu_\gamma$ is a valid Lévy measure.
Nonnegativity is required for the measure-theoretic construction of compound Poisson integrals.
\end{theorem}
\begin{proof}\leanok Product of nonneg factors: $c_\gamma \geq 0$ by hypothesis and $|z|^{-1-\gamma} \geq 0$ by \texttt{rpow\_nonneg}.\end{proof}

\section{Moment Computations}

We compute the large-jump rate and the small-jump moment integrals that determine whether the energy space is Hilbert or Banach.

\begin{theorem}[Large jump rate positive]\label{thm:large_jump_rate}
\lean{stable_large_jump_rate_pos}
\leanok
$\lambda = \nu_\gamma(\{|z| > 1\}) = 2c_\gamma/\gamma > 0$.
Positivity ensures the Poisson parameter $\lambda T > 0$, giving positive variance
for the large-jump count witness.
This rate controls the intensity of the compound Poisson component in the Lévy--Itô decomposition.
\end{theorem}
\begin{proof}\leanok Apply \texttt{div\_pos} to $2c_\gamma > 0$ (from \texttt{mul\_pos two\_pos hc}) and $\gamma > 0$ (from $\gamma > 1$).\end{proof}

\begin{theorem}[Small jump moment finite]\label{thm:small_jump_finite}
\lean{stable_small_jump_moment_finite}
\leanok
For $\alpha > \gamma$: $\int_{|z|\leq 1} |z|^\alpha\,\nu_\gamma(dz) = 2c_\gamma/(\alpha - \gamma) < \infty$.
In particular, the second moment ($\alpha = 2 > \gamma$) is finite, enabling the $L^2$ isometry
for truncated compensated integrals.
This finiteness is what makes the compensated integral well-defined in $L^2$.
\end{theorem}
\begin{proof}\leanok Apply \texttt{div\_pos} to $2c_\gamma > 0$ and $\alpha - \gamma > 0$ (from \texttt{linarith}).\end{proof}

\begin{theorem}[Small jump moment diverges]\label{thm:small_jump_diverges}
\lean{stable_small_jump_moment_diverges}
\leanok
For $\alpha < \gamma$: $\int_{|z|\leq 1} |z|^\alpha\,\nu_\gamma(dz) = \infty$.
This divergence forces the integrand space $H_L = L^p$ with $p < \gamma < 2$ into the Banach setting,
ruling out the standard Hilbert-space energy framework.
\end{theorem}
\begin{proof}\leanok The exponent $\alpha - \gamma < 0$ by \texttt{linarith}.\end{proof}

\section{Poisson Identities}

This section derives the Poisson mean, second factorial moment, and variance from a single recurrence relation.

\begin{defn}[Poisson recurrence]\label{lem:poisson_recurrence}
\lean{poissonPMFReal_succ_mul}
\leanok
$(k+1) \cdot P_r(k+1) = r \cdot P_r(k)$ for $P_r(k) = e^{-r} r^k / k!$.
This recurrence allows reindexing $\sum k P_r(k)$ into $r \sum P_r(k) = r$,
and is the key tool for proving both the mean and second factorial moment.
\end{defn}

\begin{theorem}[Poisson mean]\label{thm:poisson_mean}
\lean{poisson_mean_identity}
\leanok
\uses{lem:poisson_recurrence}
$\sum_{n=0}^{\infty} n \cdot P_r(n) = r$.
Used in the variance computation and centering of the obstruction witness.
This classical identity is derived here from the recurrence to keep the development self-contained.
\end{theorem}
\begin{proof}\leanok
Use the recurrence to show $\texttt{HasSum}(\lambda k.\; (k+1)P_r(k+1), r)$
via \texttt{simp\_rw [poissonPMFReal\_succ\_mul]}. Shift the index to get
$\texttt{HasSum}(\lambda n.\; n \cdot P_r(n), r)$ via \texttt{comp\_nat\_add},
split off the $n=0$ term (which is zero), and conclude with \texttt{tsum\_eq}.
\end{proof}

\begin{theorem}[Poisson second factorial moment]\label{thm:poisson_second_fact}
\lean{poisson_second_factorial_hassum}
\leanok
\uses{lem:poisson_recurrence}
$\sum n(n-1) P_r(n) = r^2$, proved by applying the recurrence twice.
Combined with the mean, this yields $\mathrm{Var} = r^2 + r - r^2 = r$, completing the variance calculation.
\end{theorem}
\begin{proof}\leanok
Apply the double recurrence $(k+2)(k+1)P_r(k+2) = r^2 P_r(k)$ to get
$\texttt{HasSum}(\lambda k.\; r^2 P_r(k), r^2)$, then shift by 2 and note
that the $n=0$ and $n=1$ terms vanish.
\end{proof}

\begin{theorem}[Poisson variance]\label{thm:poisson_variance}
\lean{poisson_variance_identity}
\leanok
\uses{thm:poisson_mean, thm:poisson_second_fact}
$\mathrm{Var}(\mathrm{Poisson}(r)) = r$.
Positive variance $\lambda T > 0$ implies the centered witness is nonzero, which is essential for the representability obstruction.
\end{theorem}
\begin{proof}\leanok
Decompose $(n-r)^2 P_r(n) = n(n-1)P_r(n) + (1-2r)\cdot nP_r(n) + r^2 P_r(n)$
pointwise (by \texttt{ring}). Combine three \texttt{HasSum} instances via
\texttt{HasSum.add}: the values sum to $r^2 + (1-2r)r + r^2 = r$ (by \texttt{ring}).
Use \texttt{convert} with \texttt{ext n; ring} for the summand and \texttt{ring} for the value.
\end{proof}

\begin{proof}\leanok Expand and cancel. Closed by \texttt{ring}.\end{proof}

\begin{proof}\leanok Rewrite via \texttt{quadratic\_defect\_exact}, then apply \texttt{abs\_of\_nonneg (sq\_nonneg \_)}.\end{proof}

\chapter{Lévy--Itô Framework}

This chapter constructs the jump-side stochastic calculus from first principles.
The strategy is to avoid axiomatizing the compensated Poisson integral: instead,
we build it as the $L^2$ limit of compound Poisson finite sums (which require
only elementary summation, not measure-theoretic integration). The finite
telescoping identity replaces the It\^o formula, the $L^2$ Cauchy estimate
$\varepsilon^{2-\gamma} \to 0$ guarantees convergence, and centering and isometry
are derived (not assumed) via \texttt{tendsto\_nhds\_unique}. The chapter also
develops the Poisson chaos orthogonality on the canonical space and constructs
the compound Poisson path as an explicit c\`adl\`ag function.

\section{Truncated Measures}

The stable L\'evy measure is restricted to $\{|z| \leq M\}$ to produce a finite measure suitable for compound Poisson approximation.

\begin{defn}[Truncated stable measure]\label{def:trunc_measure}
\lean{TruncatedStableMeasure}
\leanok
$\nu_\gamma^M := \nu_\gamma|_{\{|z|\leq M\}}$, the restriction to $\{|z| \leq M\}$.
This finite measure enables compound Poisson approximation: the Itô formula
reduces to a finite telescoping sum, and the full integral is recovered as $M \to \infty$.
\end{defn}

\begin{theorem}[Truncated second moment positive]\label{thm:trunc_second}
\lean{TruncatedStableMeasure.second_moment_pos}
\leanok
\uses{def:trunc_measure}
$\int_{|z|\leq M} z^2\,\nu_\gamma(dz) = 2c_\gamma M^{2-\gamma}/(2-\gamma) > 0$.
Provides the nondegenerate isometry constant for $\delta_L^M$, ensuring that the compensated integral norm is strictly positive.
\end{theorem}
\begin{proof}\leanok Apply \texttt{div\_pos} to the product $2c_\gamma \cdot M^{2-\gamma} > 0$ (from \texttt{mul\_pos} and \texttt{rpow\_pos\_of\_pos}) and the denominator $2 - \gamma > 0$ (from \texttt{linarith}).\end{proof}

\section{Finite Telescoping}

Finite telescoping identities replace the It\^o formula in the compound Poisson setting, reducing path-level calculus to algebraic sums.

\begin{theorem}[Finite telescope]\label{thm:telescope}
\lean{finite_telescope}
\leanok
$f(y_n) - f(y_0) = \sum_{i=0}^{n-1} [f(y_{i+1}) - f(y_i)]$.
Replaces stochastic integration in the compound Poisson setting, reducing path-level identities to algebraic sums.
\end{theorem}
\begin{proof}\leanok Direct application of \texttt{(Finset.sum\_range\_sub (f $\circ$ y) n).symm}.\end{proof}

\begin{theorem}[Jump linear plus remainder]\label{thm:jump_decomp}
\lean{jump_linear_plus_remainder}
\leanok
$f(v+z) - f(v) = f'(v)z + [f(v+z) - f(v) - f'(v)z]$: the linear part (odd in $z$,
detectable by $D_L$) plus the nonlinear remainder (contributing to $\Gamma_L$).
This decomposition underpins both the product rule and the chaos characterization.
\end{theorem}
\begin{proof}\leanok Closed by \texttt{ring}.\end{proof}

\begin{theorem}[Product telescope]\label{thm:product_telescope}
\lean{product_telescope}
\leanok
\uses{thm:telescope}
Telescoping for $h = ab$ gives the compound Poisson product rule: a Leibniz part
plus a defect $[a(y_{i+1}) - a(y_i)][b(y_{i+1}) - b(y_i)]$.
The defect term is the discrete analogue of the carré du champ operator.
\end{theorem}
\begin{proof}\leanok Apply \texttt{finite\_telescope} to $h(x) = a(x) \cdot b(x)$.\end{proof}

\section{Compound Poisson Integral}

The compound Poisson integral is defined as a finite random sum with compensation, avoiding measure-theoretic convergence issues.

\begin{defn}[Compound Poisson integral]\label{def:cpi}
\lean{CompoundPoissonIntegral}
\leanok
The finite random sum $\sum_{i=1}^{N} h(s_i, z_i) - \mathrm{compensator}$,
well-defined pointwise since $\nu_\gamma^M$ has finite total mass.
Finiteness of the sum avoids the convergence issues that arise for the full Lévy integral.
\end{defn}

\begin{theorem}[Random sum linear]\label{thm:random_sum_linear}
\lean{CompoundPoissonIntegral.random_sum_linear}
\leanok
\uses{def:cpi}
$\sum(ch_1 + h_2) = c\sum h_1 + \sum h_2$, from linearity of finite sums.
This ensures that the compound Poisson integral inherits linearity from ordinary summation.
\end{theorem}
\begin{proof}\leanok Unfold \texttt{random\_sum}, apply \texttt{Finset.sum\_add\_distrib} and \texttt{Finset.mul\_sum}.\end{proof}

\section{Compensated Integral Construction}

The compensated integral $I_1^M(h)$ is built as an $L^2$ limit of compound Poisson finite sums, with centering and isometry derived by passage to the limit.

\begin{defn}[Constructed compensated integral]\label{def:cci}
\lean{ConstructedCompensatedIntegral}
\leanok
\uses{def:cpi}
$I_1^M(h) = \lim_{\varepsilon \to 0} I_1^{\varepsilon,M}(h)$ in $L^2$,
constructed from compound Poisson finite sums. Centering and isometry are
derived via \texttt{tendsto\_nhds\_unique}, transferring properties from the approximations to the limit.
\end{defn}

\begin{theorem}[$I_1$ centering from approximation]\label{thm:I1_centered}
\lean{ConstructedCompensatedIntegral.I₁_centered_from_approx}
\leanok
\uses{def:cci}
$\mathbb{E}[I_1^M(h)] = 0$: each approximation is centered,
and centering passes to the $L^2$ limit.
This zero-mean property is inherited by all divergence operators built from $I_1$.
\end{theorem}
\begin{proof}\leanok
Each $\mathbb{E}[I_1^{\varepsilon,M}(h)] = 0$ (compensated sum). The constant
sequence $0$ tends to $\mathbb{E}[I_1^M(h)]$ by \texttt{approx\_converges}.
By \texttt{tendsto\_nhds\_unique}, $\mathbb{E}[I_1^M(h)] = 0$.
\end{proof}

\begin{theorem}[$I_1$ isometry from approximation]\label{thm:I1_isometry}
\lean{ConstructedCompensatedIntegral.I₁_isometry_from_approx}
\leanok
\uses{def:cci}
$\mathbb{E}[|I_1^M(h)|^2] = \|h\|_{L^2(\nu_\gamma^M \otimes dt)}^2$:
the isometry of finite-sum approximations is preserved in the limit.
This identity is the analogue of the classical It\^o isometry [4, Theorem 4.2.3] for the compensated Poisson setting.
\end{theorem}
\begin{proof}\leanok Same pattern as centering: each approximation satisfies the isometry exactly,
so the constant sequence converges to the limit isometry by \texttt{tendsto\_nhds\_unique}.\end{proof}

\begin{defn}[Bridge: construction to axioms]\label{def:toAxioms}
\lean{ConstructedCompensatedIntegral.toAxioms}
\leanok
\uses{thm:I1_centered, thm:I1_isometry}
Packages the constructed integral into the axiomatic interface for
\texttt{LevyItoDecomposition}, converting proved properties into structure fields.
This bridge allows downstream results to use the abstract API without depending on construction details.
\end{defn}

\section{Lévy--Itô Decomposition}

The L\'evy--It\^o decomposition splits the process into compensated small jumps and compound Poisson large jumps, with truncation convergence enabling passage to the full divergence.

\begin{defn}[Lévy--Itô decomposition]\label{def:levy_ito}
\lean{LevyItoDecomposition}
\leanok
\uses{def:toAxioms}
$L_T = X_T^{\mathrm{small}} + Y_T^{\mathrm{large}}$: compensated small jumps
plus compound Poisson large jumps.
This canonical splitting separates the martingale and finite-variation components of the process.
\end{defn}

\begin{theorem}[Truncation convergence]\label{thm:trunc_conv}
\lean{LevyItoDecomposition.trunc_convergence}
\leanok
\uses{def:levy_ito}
$\mathbb{E}[F \cdot \delta^M(u)] \to \mathbb{E}[F \cdot \delta(u)]$ as $M \to \infty$.
Essential for the chaos characterization (Theorem~C), since it allows passage from truncated to full divergence.
\end{theorem}
\begin{proof}\leanok Rewrite $\delta^M(u) = I_1^M(u \cdot z)$ via \texttt{$\delta$\_trunc\_eq}, then apply \texttt{I$_1$\_convergence}.\end{proof}

\begin{theorem}[$\delta^M$ centered]\label{thm:delta_centered}
\lean{LevyItoDecomposition.δ_trunc_centered}
\leanok
\uses{def:levy_ito}
$\mathbb{E}[\delta^M(u)] = 0$, inherited from centering of $I_1^M$.
Centering of the truncated divergence is a prerequisite for the even-annihilation argument at each truncation level.
\end{theorem}
\begin{proof}\leanok Rewrite via \texttt{$\delta$\_trunc\_eq} and apply \texttt{I$_1$\_centered}.\end{proof}

\begin{theorem}[Chaos orthogonality]\label{thm:chaos_orth}
\lean{LevyItoDecomposition.chaos_orthogonality_from_kernel_vanishing}
\leanok
\uses{def:levy_ito}
If $\Phi_M \equiv 0$, then $\mathbb{E}[F \cdot \delta^M(u)] = 0$
for all deterministic $u$. Even kernels have $\Phi_M = 0$ since $f_1 \cdot z$
is odd and $\nu_\gamma$ is symmetric, connecting measure symmetry to chaos structure.
\end{theorem}
\begin{proof}\leanok Apply \texttt{chaos\_identity} with pointwise vanishing $\Phi_M(s) \cdot u(s) = 0$ from \texttt{zero\_mul}.\end{proof}

\begin{theorem}[Predictable module]\label{thm:pred_module}
\lean{LevyItoDecomposition.predictable_module_from_linearity}
\leanok
\uses{def:levy_ito}
$\mathbb{E}[(cF) \cdot \delta^M(u)] = c \cdot \mathbb{E}[F \cdot \delta^M(u)]$.
This scalar-linearity property ensures that $D_L$ extends to a module map over predictable scalars.
\end{theorem}
\begin{proof}\leanok Rewrite $cF \cdot \delta = c(F \cdot \delta) + 0$ pointwise (by \texttt{ext; simp; ring}), then apply \texttt{$\mathbb{E}$\_linear} and \texttt{$\mathbb{E}$\_const}.\end{proof}

\section{Poisson Chaos}

The Poisson inner product and centered variable $\chi$ are defined on the canonical space, and chaos orthogonality is established for the witness argument.

\begin{defn}[Poisson inner product]\label{def:poisson_inner}
\lean{poisson_inner}
\leanok
$\langle f, g \rangle_r = \sum_n f(n)\,g(n)\,P_r(n)$: the $L^2$ inner product
on $(\mathbb{N}, \mathrm{poissonMeasure}(r))$.
This discrete inner product underlies the Fourier--Charlier expansion that decomposes $L^2$ into chaos subspaces.
\end{defn}

\begin{defn}[Centered Poisson variable]\label{def:poisson_chi}
\lean{poisson_chi}
\leanok
$\chi(n) = n - r$: the first-chaos generator. The centered
witness $G_0 - \mathbb{E}[G_0]$ is $\chi$ with $r = \lambda T$.
As the simplest nonconstant orthogonal polynomial, $\chi$ spans the first Poisson chaos.
\end{defn}

\begin{theorem}[$\chi$ centered]\label{thm:chi_centered}
\lean{poisson_chi_centered}
\leanok
\uses{def:poisson_chi, thm:poisson_mean}
$\mathbb{E}[\chi] = r - r = 0$, from the Poisson mean identity.
Centering is one of the three hypotheses required by the representability obstruction.
\end{theorem}
\begin{proof}\leanok
Write $(n-r)P_r(n) = nP_r(n) - rP_r(n)$. Apply \texttt{HasSum.sub} to the
mean $\texttt{HasSum}(\ldots, r)$ and $r \cdot \texttt{HasSum}(\ldots, 1)$.
The difference is $r - r \cdot 1 = 0$ by \texttt{ring}.
\end{proof}

\begin{theorem}[$\chi$ norm squared]\label{thm:chi_norm}
\lean{poisson_chi_norm_sq}
\leanok
\uses{def:poisson_chi, thm:poisson_variance}
$\|\chi\|^2 = \mathrm{Var}(N) = r > 0$.
Implies $\chi \neq 0$ in $L^2$, which is the nonzero condition needed for the obstruction.
\end{theorem}
\begin{proof}\leanok Unfold definitions; the sum $\sum (n-r)^2 P_r(n)$ is exactly \texttt{poisson\_variance\_identity}.\end{proof}

\begin{theorem}[First chaos orthogonal]\label{thm:first_chaos}
\lean{first_chaos_orthogonal}
\leanok
\uses{def:poisson_inner, def:poisson_chi}
$\langle f, \chi \rangle = 0 \Rightarrow \langle f, a\chi \rangle = 0$ for all $a$.
Orthogonality to $\chi$ implies orthogonality to $\mathrm{span}(\chi)$, reflecting the module structure of the first chaos.
\end{theorem}
\begin{proof}\leanok Rewrite $f(n)(a\chi(n))P_r(n) = a \cdot f(n)\chi(n)P_r(n)$ by \texttt{ring}, factor out $a$ via \texttt{tsum\_mul\_left}, and multiply by zero.\end{proof}

\begin{theorem}[Chaos orthogonality concrete]\label{thm:chaos_concrete}
\lean{chaos_orthogonality_concrete}
\leanok
\uses{thm:first_chaos}
Zero first-chaos coefficient implies orthogonality to $\mathrm{span}(\chi)$.
This is the concrete form of chaos orthogonality used in the Poisson witness argument.
\end{theorem}
\begin{proof}\leanok Extract $\langle f, \chi \rangle = 0$ from the coefficient via \texttt{div\_eq\_zero\_iff}, then apply \texttt{first\_chaos\_orthogonal}.\end{proof}

\section{$L^2$ Estimates}

The $L^2$ Cauchy estimate $\varepsilon^{2-\gamma} \to 0$ and related bounds ensure convergence of the compound Poisson approximations to the compensated integral.

\begin{theorem}[$L^2$ Cauchy estimate]\label{thm:l2_cauchy}
\lean{l2_cauchy_vanishes}
\leanok
$2c_\gamma \varepsilon^{2-\gamma}/(2-\gamma) \to 0$ as $\varepsilon \to 0^+$.
Proves the approximations are $L^2$-Cauchy, guaranteeing convergence to $I_1^M(h)$.
This estimate is the analytic core of the compensated integral construction.
\end{theorem}
\begin{proof}\leanok Factor as $(2c_\gamma/(2-\gamma)) \cdot \varepsilon^{2-\gamma}$. Apply \texttt{tendsto\_const\_nhds.mul} with \texttt{rpow\_tendsto\_zero\_at\_zero\_right}.\end{proof}

\begin{theorem}[rpow tends to zero]\label{thm:rpow_zero}
\lean{rpow_tendsto_zero_at_zero_right}
\leanok
$\varepsilon^\alpha \to 0$ as $\varepsilon \to 0^+$ for $\alpha > 0$.
This basic real-analysis fact is used in the $L^2$ Cauchy estimate to control the truncation error.
\end{theorem}
\begin{proof}\leanok Use \texttt{continuous\_rpow\_const} for continuity of $x \mapsto x^\alpha$ at $0$, and \texttt{zero\_rpow} for $0^\alpha = 0$.\end{proof}

\begin{theorem}[Truncated moment bounded]\label{thm:trunc_bounded}
\lean{truncated_moment_bounded}
\leanok
$2c_\gamma M^{2-\gamma}/(2-\gamma) > 0$ for fixed $M$: the isometry constant
for $\delta_L^M$ is positive and finite.
Finiteness at each truncation level is what makes the compound Poisson approximation scheme viable.
\end{theorem}
\begin{proof}\leanok Same as \texttt{second\_moment\_pos}: \texttt{div\_pos} applied to positive numerator and positive denominator.\end{proof}

\section{Compound Poisson Path}

The compound Poisson path $Y_t = \sum_{s_i \leq t} z_i$ is constructed as an explicit c\`adl\`ag function, providing the large-jump component of the L\'evy--It\^o decomposition.

\begin{defn}[Compound Poisson path]\label{def:cp_path}
\lean{compound_poisson_path}
\leanok
\uses{def:cpi}
$Y_t = \sum_{s_i \leq t} z_i$: a pure-jump càdlàg path defined as a deterministic
function of the jump data.
This explicit path representation avoids the need for general semimartingale theory.
\end{defn}

\begin{defn}[Stable large jump path]\label{def:sljp}
\lean{StableLargeJumpPath}
\leanok
\uses{def:cpi}
Compound Poisson path restricted to $|z_i| > 1$. The jump count
$N_T \sim \mathrm{Poisson}(\lambda T)$, and the centered version $N_T - \mathbb{E}[N_T]$
is the obstruction witness that drives the main theorem.
\end{defn}

\begin{theorem}[Path starts at zero]\label{thm:path_zero}
\lean{StableLargeJumpPath.path_zero}
\leanok
\uses{def:cp_path}
$Y_0 = 0$: the sum over an empty index set vanishes.
This initial condition anchors the telescoping identity used in the Itô formula.
\end{theorem}
\begin{proof}\leanok All jump times satisfy $s_i > 0$, so each indicator $\mathbf{1}_{s_i \leq 0} = 0$. Apply \texttt{Finset.sum\_eq\_zero}.\end{proof}

\begin{theorem}[Path terminal value]\label{thm:path_terminal}
\lean{StableLargeJumpPath.path_terminal}
\leanok
\uses{def:cp_path}
$Y_T = \sum_i z_i$, since all jump times satisfy $s_i \leq T$.
At the terminal time, the path accumulates all jumps, recovering the full random sum.
\end{theorem}
\begin{proof}\leanok All indicators are $1$ by \texttt{if\_pos}. Apply \texttt{Finset.sum\_congr}.\end{proof}


\chapter{Banach Energy Space Framework}

This chapter formalizes the abstract operator factorization theory from [1].
The derivative $D = \delta^*$ is constructed (not axiomatized) via the Banach
dual pairing $\langle DF, u \rangle = \mathbb{E}[F \cdot \delta(u)]$, and
its properties---linearity, vanishing on constants, kernel characterization---are
all proved from this single identity. The main results are:
Theorem~A (fluctuation factorization on $\mathrm{im}(\delta)$),
the representability obstruction (centered even functionals cannot lie in
$\mathrm{im}(\delta_L)$, proved by the even-annihilation/self-pairing argument),
Theorem~B (the product rule with jump defect $\Gamma$), and
Theorem~C (chaos kernel characterization of $\ker(D_L)$).
The chapter concludes by instantiating the abstract framework for symmetric
$\gamma$-stable processes and verifying backward compatibility with the
Hilbert formalization via the Hilbert bridge.

\section{Abstract Framework}

The Banach energy space axiomatizes the minimal structure---divergence, duality pairings, and expectation---needed to develop the operator theory without assuming a Hilbert space identification.

\begin{defn}[Banach energy space]\label{def:banach_space}
\lean{BanachEnergySpace}
\leanok
A Banach integrand space $H_X$ with dual $H_X^*$, $L^p$--$L^q$ function spaces,
divergence $\delta : H_X \to L^p$, duality pairings, and expectation.
Unlike the Hilbert framework, no Riesz identification $H_X^* \cong H_X$ is assumed,
which is precisely the generalization required by the singular Lévy measure.
\end{defn}

\begin{defn}[Centering]\label{def:center}
\lean{BanachEnergySpace.center}
\leanok
\uses{def:banach_space}
$\mathrm{center}(G_0) = G_0 - \mathrm{const}(\mathbb{E}[G_0])$: removes the mean,
preserving variance and symmetry.
Centering is the first step in preparing the obstruction witness.
\end{defn}

\begin{theorem}[Center is centered]\label{thm:center_centered}
\lean{BanachEnergySpace.center_centered}
\leanok
\uses{def:center}
$\mathbb{E}[\mathrm{center}(G_0)] = 0$, by linearity of expectation.
One of the three hypotheses of the representability obstruction, ensuring the witness has zero mean.
\end{theorem}
\begin{proof}\leanok Expand $\mathbb{E}[G_0 - \mathrm{const}(\mathbb{E}[G_0])]$ using linearity (\texttt{expect\_sub}) and $\mathbb{E}[\mathrm{const}(c)] = c$ (\texttt{expect\_const}), giving $\mathbb{E}[G_0] - \mathbb{E}[G_0] = 0$.\end{proof}

\section{The Derivative $D$}

The operator-covariant derivative $D$ is constructed via the Banach dual pairing, and its fundamental properties---linearity, vanishing on constants, and kernel characterization---are derived from a single defining identity.

\begin{defn}[Operator-covariant derivative]\label{def:D}
\lean{BanachEnergySpace.D}
\leanok
\uses{def:banach_space}
Constructed as $D(F) := \mathrm{mk\_dual}(u \mapsto \mathbb{E}[F \cdot \delta(u)])$.
Bounded linearity on $H_X$ follows from Hölder, so $D(F) \in H_X^*$.
This operator generalizes the Malliavin derivative [3] to the non-Hilbert setting.
\end{defn}

\begin{theorem}[$D$ defining identity]\label{thm:D_def}
\lean{BanachEnergySpace.D_def}
\leanok
\uses{def:D}
$\langle DF, u \rangle = \mathbb{E}[F \cdot \delta(u)]$.
All downstream results follow from this single identity, which serves as the axiomatic interface for $D$.
\end{theorem}
\begin{proof}\leanok Unfold $D$ and apply \texttt{mk\_dual\_spec}.\end{proof}

\begin{theorem}[$D$ linear]\label{thm:D_linear}
\lean{BanachEnergySpace.D_linear}
\leanok
\uses{thm:D_def}
$D(\alpha F + G) = \alpha DF + DG$.
Linearity of $D$ is inherited from linearity of expectation and the duality pairing.
\end{theorem}
\begin{proof}\leanok Evaluate on test $u$ via the defining identity. Linearity of $\mathbb{E}$ and Hölder give equality; nondegeneracy of the pairing concludes.\end{proof}

\begin{theorem}[$D$ on constants]\label{thm:D_const}
\lean{BanachEnergySpace.D_const}
\leanok
\uses{thm:D_def}
$D(\mathrm{const}(c)) = 0$, since $\mathbb{E}[c \cdot \delta(u)] = c \cdot 0 = 0$ by (H1).
Constants carry no stochastic information, so their derivative vanishes.
\end{theorem}
\begin{proof}\leanok Apply the defining identity, factor out $c$, use \texttt{holder\_const\_left} and centering of $\delta$.\end{proof}

\begin{theorem}[$\ker(D)$ characterization]\label{thm:ker_D}
\lean{BanachEnergySpace.ker_D_iff_annihilates_im_delta}
\leanok
\uses{thm:D_def}
$DF = 0$ iff $\mathbb{E}[F \cdot \delta(u)] = 0$ for all $u$: the kernel is the
annihilator of $\mathrm{im}(\delta)$.
This characterization reduces kernel membership to a testable orthogonality condition.
\end{theorem}
\begin{proof}\leanok Forward: $DF = 0$ means $\langle DF, u \rangle = 0$ for all $u$; apply defining identity. Reverse: the pairing is zero for all $u$; apply \texttt{pair\_nondegenerate}.\end{proof}

\section{Fluctuation Factorization (Theorem A)}

Every centered functional is shown to be representable as a divergence, providing the factorization that the derivative $D$ inverts.

\begin{theorem}[Fluctuation factorization]\label{thm:factorization}
\lean{BanachEnergySpace.fluctuation_factorization}
\leanok
\uses{def:D, def:banach_space}
\textbf{Theorem~A.} Under (H3): $F - \mathbb{E}[F] = \delta(\delta^{-1}(F - \mathbb{E}[F]))$.
Every centered functional is representable as a divergence, providing the factorization that $D$ inverts.
\end{theorem}
\begin{proof}\leanok By hypothesis $F - \mathbb{E}[F] \in \mathrm{im}(\delta)$, so the recovery map gives $v$ with $\delta(v) = F - \mathbb{E}[F]$. Apply \texttt{recovery\_map\_spec}.\end{proof}

\begin{theorem}[Duality recovery]\label{thm:duality_recovery}
\lean{BanachEnergySpace.duality_recovery}
\leanok
\uses{thm:factorization, thm:D_def}
$\langle DF, u \rangle = \mathbb{E}[\delta(v) \cdot \delta(u)]$ where $v = \delta^{-1}(F - \mathbb{E}[F])$.
This expresses the derivative purely in terms of the divergence operator, closing the duality loop.
\end{theorem}
\begin{proof}\leanok Apply the defining identity $\langle DF, u \rangle = \mathbb{E}[F \cdot \delta(u)]$.\end{proof}

\section{Representability Obstruction}

The even-annihilation argument shows that centered, even, nonzero functionals cannot lie in $\mathrm{im}(\delta_L)$, providing the abstract engine for the main non-representability theorem.

\begin{defn}[Distributional symmetry]\label{def:dist_sym}
\lean{BanachEnergySpace.DistributionalSymmetry}
\leanok
\uses{def:banach_space}
$L \stackrel{d}{=} -L$: the negation map preserves even functionals while flipping $\delta_L(u)$.
This symmetry is the structural reason that even functionals are invisible to the first chaos.
\end{defn}

\begin{theorem}[Even annihilates $\mathrm{im}(\delta)$]\label{thm:even_annihilates}
\lean{BanachEnergySpace.even_annihilates_im_delta}
\leanok
\uses{def:dist_sym, thm:D_def}
If $G$ is even, then $\mathbb{E}[G \cdot \delta(u)] = 0$ for all $u$.
This vanishing is the mechanism by which distributional symmetry produces the obstruction.
\end{theorem}
\begin{proof}\leanok Under $L \mapsto -L$: $G$ is invariant, $\delta(u)$ flips sign. So $\mathbb{E}[G \cdot \delta(u)] = \mathbb{E}[G \cdot (-\delta(u))] = -\mathbb{E}[G \cdot \delta(u)]$, forcing the value to $0$. Uses \texttt{holder\_neg} and \texttt{sym.neg\_delta}.\end{proof}

\begin{theorem}[Representability obstruction]\label{thm:obstruction}
\lean{BanachEnergySpace.representability_obstruction}
\leanok
\uses{thm:even_annihilates, thm:center_centered}
\textbf{Proposition 5.1.} Centered, even, nonzero $G \Rightarrow G \notin \mathrm{im}(\delta_L)$.
This is the abstract engine of the main theorem [1, Proposition 5.1]: three checkable properties yield non-representability.
\end{theorem}
\begin{proof}\leanok By contradiction: assume $G = \delta(u)$. Then $\|G\|_2^2 = \mathbb{E}[G \cdot G] = \mathbb{E}[G \cdot \delta(u)] = 0$ by even annihilation. So $G = 0$, contradicting $G \neq 0$.\end{proof}

\section{Leibniz Defect (Theorem B)}

The Leibniz defect $\Gamma(F,G)$ measures the failure of the product rule for $D$, which is shown to be entirely determined by the jump structure of the driving process.

\begin{defn}[Leibniz defect]\label{def:leibniz}
\lean{BanachEnergySpace.leibniz_defect}
\leanok
\uses{def:D}
$\Gamma(F,G) := D(FG) - F \cdot DG - G \cdot DF \in H_X^*$: measures failure of the product rule.
$\Gamma = 0$ iff $D$ is a derivation, which holds precisely when the driving process has no jumps.
\end{defn}

\begin{theorem}[Product rule with jump defect]\label{thm:product_rule}
\lean{BanachEnergySpace.product_rule_with_jump_defect}
\leanok
\uses{def:leibniz}
\textbf{Theorem~B.}
$\langle \Gamma(F,G), u \rangle = \mathbb{E}[\sum_s \Delta_s(f,g) \cdot \delta(u)]$.
The defect is entirely determined by the jump structure of the process [1, Theorem B].
\end{theorem}
\begin{proof}\leanok Apply the Itô formula to $fg$, multiply by $\delta(u)$, take expectations. The Leibniz part cancels, leaving the jump residual $\sum_s \Delta_s(f,g)$. Uses \texttt{product\_ito\_paired\_derived}.\end{proof}

\begin{theorem}[Defect vanishes without jumps]\label{thm:defect_vanishes}
\lean{BanachEnergySpace.defect_vanishes_without_jumps}
\leanok
\uses{def:leibniz}
No jumps $\Rightarrow$ $\Gamma = 0$: the classical product rule for continuous processes.
This recovers the Hilbert setting as a special case, confirming consistency with the companion formalization.
\end{theorem}
\begin{proof}\leanok If $\sum_s \Delta_s(f,g) = 0$, the product rule identity reduces to $\langle \Gamma, u \rangle = 0$ for all $u$, giving $\Gamma = 0$ by nondegeneracy.\end{proof}

\section{Chaos Characterization (Theorem C)}

The kernel of $D_L$ is characterized as the space of functionals whose chaos projections vanish, showing that even kernels and higher chaoses make $\ker(D_L)$ infinite-dimensional.

\begin{theorem}[Chaos kernel characterization]\label{thm:chaos_char}
\lean{BanachEnergySpace.chaos_kernel_characterization}
\leanok
\uses{thm:D_def}
\textbf{Theorem~C.} $D_L F = 0$ on deterministic integrands iff $\Phi_M \to 0$ as $M \to \infty$.
Even kernels and higher chaoses satisfy the vanishing condition, making $\ker(D_L)$ infinite-dimensional.
This characterization shows that the kernel of $D_L$ captures precisely the chaos-orthogonal functionals [1, Theorem C]; cf.\ [9, 10] for the Poisson chaos expansion.
\end{theorem}
\begin{proof}\leanok The $L^2$ isometry reduces $\mathbb{E}[F \cdot \delta_L^M(u)] = \int u \cdot \Phi_M\,ds$. Truncation convergence passes $M \to \infty$. The conclusion follows from nondegeneracy of the $L^q$-$L^p$ pairing.\end{proof}
